var HxProcess = HxClass.extend({
    /* @constructor
     * @method init
     * Extends <a href="class.html">HxClass</a>
     *
     * Creates three streams per process: STDIN, STDOUT, and STDERR
     * @param {Object} opts Options dictionary

    init: function(opts) {
        opts = opts || {};
        this.name = opts.name || HxGUID.next();

        this._super(opts);

        this.fd = [
            new HxStream({}),
            new HxStream({}),
            new HxStream({})
        ];
    }
